# clone은 맨 처음에만 다음부터는 pull
git clone https://github.com/jinn0135/DA28_pj03_3.git # 맨 처음에 레퍼지토리 받아올 때

git pull https://github.com/jinn0135/DA28_pj03_3.git # 다른 사람 수정사항 받아올 때
git branch -a		# 브랜치 확인
git checkout 브랜치명 	# 자기 브랜치로 이동
git status 		# 현재 작업공간 상황 확안
git add 파일명 		# 파일 생성 또는 업데이트
git rm 파일명		# 파일 삭제
git commit -m 'commit message' # 커밋 및 커밋메시지 작성
git push origin 브랜치명
